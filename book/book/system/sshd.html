Nomad

SSHd job file

job "sshd" {

  datacenters = ["falkenstein"]
  type = "system"

  group "sync" {

    volume "shared-data" {
      type            = "host"
      source          = "shared-data"
      read_only       = false
    }

    network {
      mode = "bridge"
      port "ssh" {
        host_network = "nomad-clients"
        to = 2222
      }
    }

    service {
      name     = "sshd"
      port     = "ssh"
      provider = "nomad"
    }

    task "sshd" {

      driver = "docker"

      env {
        PUID = "1000"
        PGID = "1000"
        UMASK = "0077"
        # Overwritten in prepare script
        SUDO_ACCESS = "false"
        TZ = "Europe/Berlin"
        USER_NAME = "user"
        PUBLIC_KEY_FILE = "/secrets/authorized_keys"
      }

      config {
        image = "linuxserver/openssh-server"
        ports = ["ssh"]
        volumes = [
          "ssh:/config/.ssh",
          "custom-init-d:/custom-cont-init.d",
          "usr-local-sbin:/usr/local/sbin",
        ]
      }

      logs {
        max_files     = 10
        max_file_size = 15
      }

      template {
        data = <<EOH
#~ Install additinal packages

apk add acl findutils

#~ Create ACL file

cat <<EOF> /shared_data_acl
# file: /shared-data
# owner: user
# group: users
user::rwx
group::---
other::---
default:user::rwx
default:group::---
default:other::---
EOF

#~ Apply ACL
cd /
setfacl --restore=/shared_data_acl

#~ Allow user to run sudo cmd

echo 'user ALL=(ALL) NOPASSWD: /usr/local/sbin/permfix *' >> /etc/sudoers.d/user
EOH
        destination = "custom-init-d/prepare"
        perms = "755"
      }

      template {
        data = <<EOH
#!/bin/bash

if [ $# -ne 1 ]; then
  exit 1
fi

if [[ ! -z "$1" ]]; then
  if ! echo "$1" | base64 -d > /dev/null; then
    exit 1
  fi
  OBJ="$(echo "$1" | base64 -d)"
  if [[ $(readlink  -f "$OBJ") =~ ^/shared-data/ ]]; then
    if [ -x "$OBJ" ]; then
      /bin/chmod 700 "$OBJ"
    else
      /bin/chmod 600 "$OBJ"
    fi
    /bin/chown -R {{ env "PUID" }}:{{ env "PGID" }} "$OBJ"
  fi
fi
EOH
        destination = "usr-local-sbin/permfix"
        perms = "755"
      }

      template {
        data = <<EOH
{{ with nomadVar "nomad/jobs/system/sync/sshd" }}{{ .id_ed25519_pub }}{{ end }}
EOH
        destination = "secrets/authorized_keys"
        change_mode   = "restart"
      }

      volume_mount {
        volume      = "shared-data"
        destination = "/shared-data"
      }
    }
  }
}
