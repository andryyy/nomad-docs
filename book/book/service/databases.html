Nomad

Databases & caches job file

job "databases" {
  datacenters = ["falkenstein"]
  type = "service"

  group "cache" {
    count = 1

    restart {
      attempts = 5
      interval = "1m"
      delay = "10s"
      mode = "fail"
    }

    network {
      mode = "bridge"
      port "redis" {
        host_network = "nomad-clients"
        to = 6379
      }
    }

    service {
      name     = "redis"
      port     = "redis"
      provider = "nomad"
      check {
        name     = "redis_probe"
        type     = "tcp"
        interval = "10s"
        timeout  = "2s"
      }
    }

    task "redis" {
      driver = "docker"
      env {
        TZ = "Europe/Berlin"
      }
      config {
        image = "redis:latest"
        ports = ["redis"]
      }
    }
  }

  group "index" {
    count = 1

    restart {
      attempts = 5
      interval = "1m"
      delay = "10s"
      mode = "fail"
    }

    network {
      mode = "bridge"
      port "sonic" {
        host_network = "nomad-clients"
        to = 1491
      }
    }

    service {
      name     = "sonic"
      port     = "sonic"
      provider = "nomad"
      check {
        name     = "sonic_probe"
        type     = "tcp"
        interval = "10s"
        timeout  = "2s"
      }
    }

    volume "shared-data" {
      type = "host"
      source = "shared-data"
      read_only = false
    }

    task "sonic" {
      driver = "docker"
      env {
        TZ = "Europe/Berlin"
      }
      config {
        image = "valeriansaliou/sonic:latest"
        ports = ["sonic"]
        command = "sonic"
        args = ["-c", "/etc/sonic/sonic.cfg"]
        volumes = [
          "sonic:/etc/sonic",
        ]
      }
      volume_mount {
        volume      = "shared-data"
        destination = "/shared-data"
      }
      user = "1000:1000"
      template {
        data = <<EOF
{{ with nomadVar "nomad/jobs/databases/index/sonic" }}{{ .sonic_env }}{{ end }}
EOF
        destination = "local/file.env"
        perms = "600"
        uid = 1000
        gid = 1000
        env = true
      }
      template {
        data = <<EOF
[server]
log_level = "info"

[channel]
inet = "0.0.0.0:1491"
tcp_timeout = 300
auth_password = "{{ `${env.SONIC_SECRET}` }}"

[channel.search]
query_limit_default = 50
query_limit_maximum = 200
query_alternates_try = 4

suggest_limit_default = 5
suggest_limit_maximum = 20

list_limit_default = 100
list_limit_maximum = 500

[store]
[store.kv]
path = "/shared-data/sonic/store/kv/"
retain_word_objects = 1000

[store.kv.pool]
inactive_after = 1800

[store.kv.database]
flush_after = 500
compress = true
parallelism = 4
max_files = 100
max_compactions = 1
max_flushes = 1
write_buffer = 32768
write_ahead_log = true

[store.fst]
path = "/shared-data/sonic/store/fst/"

[store.fst.pool]
inactive_after = 300

[store.fst.graph]
consolidate_after = 180
max_size = 2048
max_words = 250000
EOF
        destination = "sonic/sonic.cfg"
      }
    }
  }

  group "mariadb" {
    count = 1

    volume "shared-data" {
      type = "host"
      source = "shared-data"
      read_only = false
    }

    network {
      mode = "bridge"
      port "mariadb" {
        host_network = "nomad-clients"
        to = 3306
      }
    }

    service {
      name     = "mariadb"
      port     = "mariadb"
      provider = "nomad"
      check {
        name     = "mariadb_probe"
        type     = "tcp"
        interval = "10s"
        timeout  = "2s"
      }
    }

    task "sync-check-prestart" {
      driver = "exec"
      config {
        command = "/bin/sh"
        args    = ["-c", <<EOF
i=0
while true; do
  i=$((i+1))
  curl --output /dev/null --silent --head -m 3 --fail $MUTAGEN_HTTP && { echo "Loop $i: Mutagen is healthy."; exit 0; }
  [ $i -gt 10 ] && { echo "Loop $i: Mutagen is unhealthy, giving up."; exit 1; }
  echo "Loop $i: Mutagen is unhealthy, trying again..." ; sleep 3 ;
done
EOF
]
      }
      template {
        data = <<EOF
MUTAGEN_HTTP=http://{{ with nomadService "mutagen" }}{{ with index . 0 }}{{ .Address }}:{{ .Port }}{{ end }}{{ end }}/mutagen/health.json
EOF
        destination = "local/file.env"
        env = true
      }

      lifecycle {
        hook    = "prestart"
        sidecar = false
      }
    }

    task "sync-check-poststop" {
      driver = "exec"
      config {
        command = "/bin/sh"
        args    = ["-c", <<EOF
echo "Flushing Mutagen cache..."
echo -n "Return code: "
curl $MUTAGEN_HTTP
EOF
]
      }
      template {
        data = <<EOF
MUTAGEN_HTTP=http://{{ with nomadService "mutagen" }}{{ with index . 0 }}{{ .Address }}:{{ .Port }}{{ end }}{{ end }}/mutagen/flush
EOF
        destination = "local/file.env"
        env = true
      }

      lifecycle {
        hook    = "poststop"
        sidecar = false
      }
    }

    task "mariadb" {
      driver = "docker"
      env {
        PUID = "1000"
        PGID = "1000"
        UMASK = "0066"
        TZ = "Europe/Berlin"
        DATADIR = "/shared-data/mariadb"
      }
      config {
        image = "linuxserver/mariadb"
        ports = ["mariadb"]
        volumes = [
          "config:/config",
          "custom-init-d:/custom-cont-init.d"
        ]
      }
      logs {
        max_files     = 10
        max_file_size = 15
      }
      template {
        data = <<EOH
# todo: use task name
{{ with nomadVar "nomad/jobs/databases/mariadb/mariadb" }}{{ .mariadb_env }}{{ end }}
EOH
        destination = "config/env"
        change_mode   = "restart"
      }
      volume_mount {
        volume      = "shared-data"
        destination = "/shared-data"
      }
      template {
        data = <<EOH
# Here if you need me
#/bin/sh
echo "Starting backup..."
if [ -d "/shared-data/mariadb" ]; then
  echo "Found directory"
  source /config/env
  mariabackup -uroot -p$MYSQL_ROOT_PASSWORD --backup--target-dir=/shared-data/mariadb-backup
  mariabackup -uroot -p$MYSQL_ROOT_PASSWORD --prepare --target-dir=/shared-data/mariadb-backup
fi
EOH
        destination = "custom-init-d/prepare"
        perms = "755"
      }
    }
  }
}
